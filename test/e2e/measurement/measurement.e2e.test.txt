    await beforeAllE2e();

  beforeEach(async () => {
    // Clear test data
    await TestDataSource.getRepository(MeasurementDAO).clear();
    await TestDataSource.getRepository(SensorDAO).clear();
    await TestDataSource.getRepository(GatewayDAO).clear();
    await TestDataSource.getRepository(NetworkDAO).clear();

    jest.restoreAllMocks();

  // Helper functions for test setup
  async function setupMeasurementTestData(networkCode: string, gatewayMac: string, sensorMac: string, includeOutlier = false) {
    // Create network
    const network = await TestDataSource.getRepository(NetworkDAO).save({
      code: networkCode
    });

    // Create gateway
    const gateway = await TestDataSource.getRepository(GatewayDAO).save({
      macAddress: gatewayMac,
      network
    });

    // Create sensor
    const sensor = await TestDataSource.getRepository(SensorDAO).save({
      macAddress: sensorMac,
      name: "Temp Sensor",
      variable: "temperature",
      unit: "C",
      gateway
    });

    // Create measurements
    await TestDataSource.getRepository(MeasurementDAO).save([
      { 
        createdAt: new Date("2023-01-01T12:00:00.000Z"), 
        value: 25.5, 
        sensor 
      },
      { 
        createdAt: new Date("2023-01-01T13:00:00.000Z"), 
        value: 26.0, 
        sensor 
      }
    ]);

    // Add outlier if requested
    if (includeOutlier) {
      await TestDataSource.getRepository(MeasurementDAO).save({
        createdAt: new Date("2023-01-01T14:00:00.000Z"),
        value: 100.0, // This should be detected as outlier
        sensor
      });
    }

    return { network, gateway, sensor };
  }

  async function setupNetworkWithMultipleSensors(networkCode: string, gatewayMac: string, sensorMacs: string[], includeOutlier = false) {
    // Create network
    const network = await TestDataSource.getRepository(NetworkDAO).save({
      code: networkCode
    });

    // Create gateway
    const gateway = await TestDataSource.getRepository(GatewayDAO).save({
      macAddress: gatewayMac,
      network
    });

    // Create sensors and measurements for each
    const sensors = [];
    for (let i = 0; i < sensorMacs.length; i++) {
      const sensorMac = sensorMacs[i];
      
      const sensor = await TestDataSource.getRepository(SensorDAO).save({
        macAddress: sensorMac,
        name: `Sensor ${i+1}`,
        variable: i === 0 ? "temperature" : "humidity",
        unit: i === 0 ? "C" : "%",
        gateway
      });
      
      sensors.push(sensor);

      // Create base measurements
      await TestDataSource.getRepository(MeasurementDAO).save([
        { 
          createdAt: new Date("2023-01-01T12:00:00.000Z"), 
          value: 25.0 + i, 
          sensor 
        },
        { 
          createdAt: new Date("2023-01-01T13:00:00.000Z"), 
          value: 26.0 + i, 
          sensor 
        }
      ]);

      // Add outlier if requested
      if (includeOutlier) {
        await TestDataSource.getRepository(MeasurementDAO).save({
          createdAt: new Date("2023-01-01T14:00:00.000Z"),
          value: 100.0 + i, // This should be detected as outlier
          sensor
        });
      }
    }
    
    return { network, gateway, sensors };
  }

      code: NET

    const gateway = await TestDataSource.getRepository(GatewayDAO).save({
      macAddress: GAT,

      name: "Temp Sensor",
      variable: "temperature",
      unit: "C",
      gateway
    });
    
    return { network, gateway, sensor };

        .get(`/api/v1/networks/${NET}/gateways/${GAT}/sensors/${SENSOR}/measurements`)

      const res = await request(app)
        .get(`/api/v1/networks/${NET}/gateways/${GAT}/sensors/${SENSOR}/measurements`)

          endDate: END_DATE 
        })

      expect(res.body.error).toBe(BadRequest);

      // Create a measurement from 3 days ago (outside our range)
      const threeDaysAgo = new Date();
      threeDaysAgo.setDate(threeDaysAgo.getDate() - 3);
      
      const { sensor } = await setupBasicEntities();
      
      await TestDataSource.getRepository(MeasurementDAO).save({
        createdAt: threeDaysAgo,
        value: 18.0,
        sensor
      });
      
      // Define date range that only includes recent measurements
      const startDate = new Date();
      startDate.setDate(startDate.getDate() - 2); // 2 days ago
      const endDate = new Date();
      
      const res = await request(app)
        .get(`/api/v1/networks/${NET}/gateways/${GAT}/sensors/${SENSOR}/measurements`)
        .query({ 
          startDate: startDate.toISOString(),
          endDate: endDate.toISOString() 
        })
        .set('Authorization', `Bearer ${tokenAdmin}`);
      
      expect(res.status).toBe(200);
      expect(res.body).toHaveProperty('sensorMacAddress', SENSOR);
      expect(res.body.measurements.length).toBeGreaterThan(0);
    });
  });
